macro_rules! shl_imm_u16_m128i {
    ($a:expr, $imm:expr) => { ... };
}

Shifts all u16 lanes left by an immediate.

let a = m128i::from([1_u16, 2, 3, 4, 1, 2, 3, 4]);
let c: [u16; 8] = shl_imm_u16_m128i!(a, 3).into();
assert_eq!(
  c,
  [1_u16 << 3, 2 << 3, 3 << 3, 4 << 3, 1 << 3, 2 << 3, 3 << 3, 4 << 3]
);
